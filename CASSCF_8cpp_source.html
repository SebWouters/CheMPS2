200 void CheMPS2::CASSCF::rotate2DMand1DM(const int N, double * eigenvecs, double * work){
201 
202  char notr = 'N';
203  char tran = 'T';
204  double alpha = 1.0;
205  double beta = 0.0;
206 
207  int power1 = nOrbDMRG;
208  int power2 = nOrbDMRG*nOrbDMRG;
209  int power3 = nOrbDMRG*nOrbDMRG*nOrbDMRG;
210 
211  //2DM: Gamma_{ijkl} --> Gamma_{ajkl}
212  dgemm_(&tran,&notr,&power1,&power3,&power1,&alpha,eigenvecs,&power1,DMRG2DM,&power1,&beta,work,&power1);
213  //2DM: Gamma_{ajkl} --> Gamma_{ajkd}
214  dgemm_(&notr,&notr,&power3,&power1,&power1,&alpha,work,&power3,eigenvecs,&power1,&beta,DMRG2DM,&power3);
215  //2DM: Gamma_{ajkd} --> Gamma_{ajcd}
216  for (int cnt=0; cnt<nOrbDMRG; cnt++){
217  dgemm_(&notr,&notr,&power2,&power1,&power1,&alpha,DMRG2DM + cnt*power3,&power2,eigenvecs,&power1,&beta,work + cnt*power3,&power2);
218  }
219  //2DM: Gamma_{ajcd} --> Gamma_{abcd}
220  for (int cnt=0; cnt<power2; cnt++){
221  dgemm_(&notr,&notr,&power1,&power1,&power1,&alpha,work + cnt*power2,&power1,eigenvecs,&power1,&beta,DMRG2DM + cnt*power2,&power1);
222  }
223 
224  //Update 1DM
225  setDMRG1DM(N);
226

247 void CheMPS2::CASSCF::constructCoulombAndExchangeMatrixInOrigIndices(double ** densityMatrix, double ** result){
248 
249  for (int irrepQ=0; irrepQ<numberOfIrreps; irrepQ++){
250 
251  const int linearsizeQ = iHandler->getNORB(irrepQ);
252  const int numberOfUniqueIndices = (linearsizeQ * (linearsizeQ + 1))/2;
253 
254  #pragma omp parallel for schedule(static)
255  for (int combinedindex=0; combinedindex<numberOfUniqueIndices; combinedindex++){
256 
257  int colQ = 1;
258  while ( (colQ*(colQ+1))/2 <= combinedindex ){ colQ++; }
259  colQ -= 1;
260  int rowQ = combinedindex - (colQ*(colQ+1))/2;
261 
262  const int HamIndexI = iHandler->getOrigNOCCstart(irrepQ) + rowQ;
263  const int HamIndexK = iHandler->getOrigNOCCstart(irrepQ) + colQ;
264 
265  double * myPtr = result[irrepQ] + rowQ + linearsizeQ * colQ;
266  myPtr[0] = 0.0;
267 
268  for (int irrepN=0; irrepN<numberOfIrreps; irrepN++){
269  const int linearsizeN = iHandler->getNORB(irrepN);
270  for (int rowN=0; rowN<linearsizeN; rowN++){
271  const int HamIndexJ = iHandler->getOrigNOCCstart(irrepN) + rowN;
272 
273  myPtr[0] += densityMatrix[irrepN][rowN*(1+linearsizeN)] * ( HamOrig->getVmat(HamIndexI,HamIndexJ,HamIndexK,HamIndexJ)
274  - 0.5 * HamOrig->getVmat(HamIndexI,HamIndexK,HamIndexJ,HamIndexJ) );
275 
276  for (int colN = rowN+1; colN<linearsizeN; colN++){
277  const int HamIndexL = iHandler->getOrigNOCCstart(irrepN) + colN;
278 
279  myPtr[0] += densityMatrix[irrepN][rowN + linearsizeN * colN] *
280  ( HamOrig->getVmat(HamIndexI,HamIndexJ,HamIndexK,HamIndexL) - 0.5 * HamOrig->getVmat(HamIndexI,HamIndexK,HamIndexJ,HamIndexL)
281  + HamOrig->getVmat(HamIndexI,HamIndexL,HamIndexK,HamIndexJ) - 0.5 * HamOrig->getVmat(HamIndexI,HamIndexK,HamIndexL,HamIndexJ) );
282 
283  }
284  }
285  }
286 
287  result[irrepQ][colQ + linearsizeQ * rowQ] = myPtr[0];
288 
289  }
290  }
